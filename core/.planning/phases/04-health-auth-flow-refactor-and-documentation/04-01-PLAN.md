enum class ConnectionStatus {
    CONNECTED, SYNCING, HEALTHY, DEGRADED, STALE, DISCONNECTING, DISCONNECTED, FAILED;

    fun canTransitionTo(newStatus: ConnectionStatus): Boolean {
        return when (this) {
            CONNECTED     -> newStatus in setOf(SYNCING, HEALTHY, DISCONNECTING, FAILED)
            SYNCING       -> newStatus in setOf(HEALTHY, DEGRADED, FAILED)
            HEALTHY       -> newStatus in setOf(SYNCING, STALE, DEGRADED, DISCONNECTING, FAILED)
            DEGRADED      -> newStatus in setOf(HEALTHY, STALE, FAILED, DISCONNECTING) // MISSING: SYNCING
            STALE         -> newStatus in setOf(SYNCING, DISCONNECTING, FAILED)
            DISCONNECTING -> newStatus in setOf(DISCONNECTED, FAILED)
            DISCONNECTED  -> newStatus in setOf(CONNECTED)
            FAILED        -> newStatus in setOf(CONNECTED, DISCONNECTED) // MISSING: SYNCING
        }
    }
}

    - All existing transitions unchanged (regression safety)

    IntegrationHealthService.evaluateConnectionHealth(connectionId):
    - All SUCCESS sync states -> connection transitions to HEALTHY
    - Any entity type with consecutiveFailureCount >= 3 -> connection transitions to DEGRADED
    - All FAILED sync states -> connection transitions to FAILED
    - DEGRADED check takes priority over FAILED (if any has 3+ failures, result is DEGRADED even if not all are FAILED)
    - Empty sync states list -> no-op (no status change, no save)
    - Connection not found -> no-op (graceful return)
    - Invalid transition (e.g., connection is DISCONNECTING) -> no-op + warn logged
    - Auto-recovery: connection was DEGRADED, all entity types now SUCCESS -> transitions to HEALTHY via SYNCING cycle
  </behavior>
