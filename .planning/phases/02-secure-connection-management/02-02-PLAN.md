    data class EncryptedCredentials(
        val ciphertext: ByteArray,

        val keyVersion: Int,

    **4. CredentialEncryptionService** — follow research Pattern 1 exactly:
    ```kotlin
    @Service
    class CredentialEncryptionService(
        private val logger: KLogger,
        props: CustomSourceConfigurationProperties,
    ) {
        private val secretKeySpec: SecretKeySpec = initKey(props.credentialEncryptionKey)

        private fun initKey(base64Key: String): SecretKeySpec {
            require(base64Key.isNotBlank()) {
                "RIVEN_CREDENTIAL_ENCRYPTION_KEY must be set (base64-encoded 32-byte key)"
            }
            val keyBytes = try { Base64.getDecoder().decode(base64Key) }
                catch (e: IllegalArgumentException) {
                    throw CryptoException("RIVEN_CREDENTIAL_ENCRYPTION_KEY is not valid Base64", e)
                }
            require(keyBytes.size == 32) {
                "RIVEN_CREDENTIAL_ENCRYPTION_KEY must decode to 32 bytes (256-bit); got ${keyBytes.size}"
            }
            return SecretKeySpec(keyBytes, "AES")
        }

        fun encrypt(plaintext: String): EncryptedCredentials {
            val iv = ByteArray(12).also { SecureRandom().nextBytes(it) }
            val cipher = Cipher.getInstance("AES/GCM/NoPadding")
            cipher.init(Cipher.ENCRYPT_MODE, secretKeySpec, GCMParameterSpec(128, iv))
            val ciphertext = cipher.doFinal(plaintext.toByteArray(Charsets.UTF_8))
            return EncryptedCredentials(ciphertext, iv, keyVersion = 1)
        }

        fun decrypt(encrypted: EncryptedCredentials): String {
            return try {
                val cipher = Cipher.getInstance("AES/GCM/NoPadding")
                cipher.init(Cipher.DECRYPT_MODE, secretKeySpec, GCMParameterSpec(128, encrypted.iv))
                String(cipher.doFinal(encrypted.ciphertext), Charsets.UTF_8)
            } catch (e: AEADBadTagException) {
                throw DataCorruptionException("Credential ciphertext is corrupted or key mismatch", e)
            } catch (e: GeneralSecurityException) {
                throw CryptoException("Credential decryption failed", e)
            }
        }
    }
    ```
    Do NOT log the plaintext anywhere. The KLogger is injected for future operational logging but must never receive credential material.

  <action>
    **Primary mechanism: TurboFilter (per RESEARCH.md primary recommendation).** Rationale: a `PatternConverter` only scrubs the formatted `%msg` output on specific appender pattern layouts, which means raw `logger.error(sqlException)` calls from third-party code that bypass the custom pattern — OR exception messages rendered by the default `ThrowableProxyConverter` — can still leak credentials. A `TurboFilter` intercepts the log event at the `LoggerContext` level BEFORE any appender or layout runs, and can rewrite the message + params + throwable references in-place. This is the only mechanism that satisfies the locked decision without requiring an `logback-spring.xml` that we must keep perfectly synchronized with every appender configuration.

    **1. `LogRedactionConfiguration.kt`** — houses the TurboFilter and registers it:
    ```kotlin
    package riven.core.configuration.customsource

    import ch.qos.logback.classic.Level
    import ch.qos.logback.classic.Logger
    import ch.qos.logback.classic.LoggerContext
    import ch.qos.logback.classic.turbo.TurboFilter
    import ch.qos.logback.core.spi.FilterReply
    import jakarta.annotation.PostConstruct
    import org.slf4j.LoggerFactory
    import org.slf4j.Marker
    import org.springframework.context.annotation.Configuration

    private val JDBC_URL_PATTERN = Regex("(jdbc:)?postgresql://[^\\s,;\"'<>]+")
    private val PASSWORD_PATTERN = Regex("(?i)password=[^\\s&;,<>\"']+")

    internal fun scrub(input: String?): String? {
        if (input == null) return null
        return input
            .replace(JDBC_URL_PATTERN, "[REDACTED_JDBC_URL]")
            .replace(PASSWORD_PATTERN, "password=[REDACTED]")
    }

    /**
     * Global TurboFilter that rewrites log format strings + params + throwable messages to scrub
     * credential material. Registered on the root LoggerContext so every logger in the JVM — including
     * the PostgreSQL JDBC driver, HikariCP, and Spring internals — is covered.
     *
     * TurboFilter returns NEUTRAL (never suppresses). Its only side effect is mutating the
     * format-string argument and scanning params in the caller's thread before the log framework
     * hands the event to appenders. We cannot mutate the format string in-place through the SLF4J
     * API, so the strategy is: (a) re-log the scrubbed form via the same logger at the same level,
     * (b) return DENY on the original event, breaking the recursion with a ThreadLocal re-entry guard.
     *
     * This is the standard Logback-recommended pattern for rewrite-style TurboFilters. See
     * RESEARCH.md Pattern 4 for the reasoning and citations.
     */
    class CredentialRedactionTurboFilter : TurboFilter() {
        private val inFilter = ThreadLocal.withInitial { false }

        override fun decide(
            marker: Marker?,
            logger: Logger,
            level: Level,
            format: String?,
            params: Array<out Any?>?,
            t: Throwable?,
        ): FilterReply {
            if (inFilter.get()) return FilterReply.NEUTRAL // break recursion
            if (format == null && t == null) return FilterReply.NEUTRAL

            val originalRendered = renderForInspection(format, params, t)
            val scrubbedRendered = scrub(originalRendered)
            if (originalRendered == scrubbedRendered) {
                return FilterReply.NEUTRAL // nothing sensitive; let it through unchanged
            }

            // Re-log with scrubbed content at the same level, guarding against re-entry
            inFilter.set(true)
            try {
                when (level.toInt()) {
                    Level.ERROR_INT -> logger.error(scrubbedRendered)
                    Level.WARN_INT  -> logger.warn(scrubbedRendered)
                    Level.INFO_INT  -> logger.info(scrubbedRendered)
                    Level.DEBUG_INT -> logger.debug(scrubbedRendered)
                    Level.TRACE_INT -> logger.trace(scrubbedRendered)
                }
            } finally {
                inFilter.set(false)
            }
            return FilterReply.DENY // suppress the original unscrubbed event
        }

        private fun renderForInspection(format: String?, params: Array<out Any?>?, t: Throwable?): String {
            val sb = StringBuilder()
            if (format != null) {
                // Minimal MessageFormatter-style expansion — sufficient to detect credential material.
                // Use SLF4J's MessageFormatter for fidelity:
                sb.append(org.slf4j.helpers.MessageFormatter.arrayFormat(format, params).message)
            }
            if (t != null) {
                var cur: Throwable? = t
                while (cur != null) {
                    sb.append(' ').append(cur.message ?: "")
                    cur = cur.cause
                }
            }
            return sb.toString()
        }
    }

    @Configuration
    class LogRedactionConfiguration {

        @PostConstruct
        fun registerRedactionFilter() {
            val loggerContext = LoggerFactory.getILoggerFactory() as LoggerContext
            // Idempotent: only add if not already present (guards test context reloads)
            if (loggerContext.turboFilterList.none { it is CredentialRedactionTurboFilter }) {
                loggerContext.addTurboFilter(CredentialRedactionTurboFilter())
            }
        }
    }
    ```

    **Tradeoff documented (per checker requirement):** The TurboFilter approach has ONE known limitation — it loses the `Throwable` stack trace on re-log (we replay only a scrubbed `String` message). Full stack traces to the error level are still visible via `logger.error(msg, t)` originating OUTSIDE this filter's re-entry guard. For Phase 2 this is acceptable: our own connection-service code wraps/sanitizes exceptions before logging (primary defense), and the TurboFilter is belt-and-suspenders for the third-party path where we don't control the caller. If Phase 6 health surfaces want stack-preserving redaction, we revisit with a custom `ThrowableConverter` in `logback-spring.xml` as a supplementary mechanism. Document this in the plan's SUMMARY.

    @SpringBootTest(classes = [LogRedactionConfiguration::class])
    class LogRedactionTest {

        private lateinit var captureAppender: ListAppender<ILoggingEvent>
        private lateinit var rootLogger: ch.qos.logback.classic.Logger

        @BeforeEach
        fun attachAppender() {
            val ctx = LoggerFactory.getILoggerFactory() as LoggerContext
            rootLogger = ctx.getLogger(Logger.ROOT_LOGGER_NAME)
            captureAppender = ListAppender<ILoggingEvent>().apply { context = ctx; start() }
            rootLogger.addAppender(captureAppender)
        }

        @AfterEach
        fun detachAppender() { rootLogger.detachAppender(captureAppender) }

        @Test
        fun `scrubs jdbc postgresql URL in info message`() {
            val log = LoggerFactory.getLogger("test.direct")
            log.info("connecting to jdbc:postgresql://user:pw@db.example.com:5432/mydb")
            val msgs = captureAppender.list.map { it.formattedMessage }
            assertTrue(msgs.any { it.contains("[REDACTED_JDBC_URL]") })
            assertTrue(msgs.none { it.contains("pw@db.example.com") })
        }

        @Test
        fun `scrubs password query param`() {
            val log = LoggerFactory.getLogger("test.pwd")
            log.info("Error with password=hunter2 in payload")
            val msgs = captureAppender.list.map { it.formattedMessage }
            assertTrue(msgs.any { it.contains("password=[REDACTED]") })
            assertTrue(msgs.none { it.contains("hunter2") })

        // REQUIRED by checker: simulates third-party logger.error(msg, sqlException) path

                msgs.none { it.contains("jdbc:postgresql://leak") },
