            ${columns.joinToString("\n") { "- ${it.name} (${it.pgType}, ${if (it.nullable) "nullable" else "not null"}${if (it.isPrimaryKey) ", primary key" else ""}${if (it.isForeignKey) ", foreign key to ${it.fkTarget}" else ""})" }}
