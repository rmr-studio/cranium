```kotlin
override fun generateSignedUrl(key: String, expiresIn: Duration): String {
    try {
        val request = GetObjectRequest { bucket = bucketName; this.key = key }
        val presigned = runBlocking { s3Client.presignGetObject(request, expiresIn.toKotlinDuration()) }
        return presigned.url.toString()
    } catch (e: Exception) {
        throw StorageProviderException("Signed URL generation failed for key: $key", e)
    }
}
```
